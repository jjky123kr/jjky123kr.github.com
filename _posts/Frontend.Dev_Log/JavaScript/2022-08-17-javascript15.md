### login 유효성 검사

ID의 Id 값과, passwd 의 Id값을 가져와야한다. 
getElementById는 id의 값을 가져온다. 
`````html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">

<title>로그인 폼</title>
<script>

function check(){
	var id = document.getElementById("id");
	var passwd = document.getElementById("passwd");
	
	if(id.value==""){
		alert("ID를 입력하세요");
		id.focus();
		return false;
	}
	
	if(passwd.value==""){
		alert("passwd 를 입력하세요");
		passwd.focus();
		return false;
	}
} 
</script>

<!-- 외부자바스크립스트 파일 불러오기 -->

<script src="login.js"></script>




</head>
<body>

autofocus: 
required: 반드시 입력 해야한다.
id : input 태그가 여러가지 사용했을때 태그를 구분해주는 역할(유효성 검사할 때)
````````

`````html
function check(){
	var writer = document.getElementById("writer")
	var passwd = document.getElementById("passwd")
	var subject = document.getElementById("subject")
	var content = document.getElementById("content")
	
	
	if(writer.value==""){
		alert("작성자명을 입력하세요")
		writer.focus();
		return false;
	}

    if(passwd.value==""){
    	alert("비밀번호를 입력하세요")
    	passwd.focus();
    	return false;
    }	
    if(passwd.value.length<2 || passwd.value.length>8){
    	alert("비밀번호는 2~8자 이내로  입력하세요");
    	passwd.value="";
    	passwd.focus();
    	return false;
    }
    
    if(subject.value==""){
    	alert("제목을 입력하세요")
    	subject.focus();
    	return false;
    }
    
    if(subject.value.length>50){
    	alert("50자 이내로 입력하세요")
    	subject.value==""
    	subject.focus();
    	return false;
    }
    if(content.value==""){
    	alert("내용을 입력하세요")
       content.focus();
    	return false;
 }
    if(content.value.length>200){
    	alert("내용을 200이내로 입력하세요")
    	content.focus();
    }
    
    
    }
``````

``````````html
// ID중복 검사
	function idcheck(){		
		var id = document.getElementById("id");
		if(id.value == ""){
			alert("ID를 입력 하세요.");
			id.focus();
			return false;
		}else{
			var ref = "idcheck.jsp?id="+id.value;	
			window.open(ref,"idcheck","width=300,height=200");
		}
	}

	// 주민번호 뒷자리로 포커스 이동
	function move(){
		var jumin1 = document.getElementById("jumin1");
		var jumin2 = document.getElementById("jumin2");
		
		if(jumin1.value.length == 6)  // 주민번호 앞자리를 6자리 입력 했으면
			jumin2.focus();
	}
	
	// 도메인 선택
	function sel(){
		var domain = document.getElementById("domain");
		var email = document.getElementById("email");
		
		if(email.value == ""){			// '직접입력' 을 선택
			domain.readOnly = false;	// 읽기 전용 해제
			domain.value = "";
		    domain.focus();
		}else{							// 도메인을 선택
			domain.value = email.value;
		    domain.readOnly = true;		// 읽기만 가능함.
		}		
	}
	
	// 유효성 검사
	function check(){
````````````
